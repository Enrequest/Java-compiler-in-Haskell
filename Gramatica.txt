
G                      -> "class" Identificador "{" Miembros "}"
Miembros               -> Miembro ";" Miembros
                        | Epsilon
Miembro                -> Tipo Identificador 
                     |Tipo Identificador "(" Parametros ")" Cuerpo
Parametros             -> Parametro
                        | Parametro "," Parametros
                        | Epsilon
Parametro              -> Tipo Identificador
Cuerpo                 -> Bloque | ";"
Bloque                 -> "{" DeclaracionesBloque "}"|"{" "}"
DeclaracionesBloque    -> DeclaracionBloque | DeclaracionBloque DeclaracionesBloque

DeclaracionBloque      -> DeclaracionVariableLocal
                        | Declaracion
DeclaracionVariableLocal -> VariableLocal ";"
VariableLocal          -> Tipo DeclaradoresVariables
DeclaradoresVariables  -> DeclaradorVariable
                        | DeclaradorVariable ","  DeclaradoresVariables
DeclaradorVariable     -> Indentificador
                        | Indentificador "=" InizializadorVariable
Declaracion            -> DeclaracionExpresion ";"
DeclaracionExpresion   -> Asignacion
                        | ExpresionPreincremento
                        | ExpresionPostincremento
                        | ExpresionPredecremento
                        | ExpresionPostdecremento
                        | InvocacionMetodo
Asignacion             -> Identificador OperadorAsignacion InizializadorVariable
OperadorAsignacion     -> "="| "*="|"/="|"%="|"+="|"-="
InizializadorVariable  -> Literal
                        | Identificador
                        
Literal                -> IntegerLit

{--

ExpresionAsignacion    -> ExpresionRelacional
                        | Asignacion
                        | Identificador
ExpresionRelacional    -> ExpresionAditiva 
                        | ExpresionMultiplicativa
--}
